with ranked_matches as (
    select  c.subject_id,
            c.charttime as anchor_time,
            d.dicom_id, 
            replace(d.path_image, '/home/andy/Documenti/Tesi-Magistrale/', '') as path_image,
            
            -- Calcoliamo il timestamp dell'RX sommando i secondi alla data
            (d.formatted_date::timestamp + (d.StudyTime * INTERVAL '1 second')) as rx_time,
            
            -- Calcoliamo la differenza in ore usando il timestamp appena creato
            abs(extract(epoch from ((d.formatted_date::timestamp + (d.StudyTime * INTERVAL '1 second')) - c.charttime))) / 3600.0 as time_diff,
            
            c.target,
            row_number() over (
                partition by c.subject_id, c.charttime 
                order by abs(extract(epoch from ((d.formatted_date::timestamp + (d.StudyTime * INTERVAL '1 second')) - c.charttime))) asc
            ) as rank
    from clinical c
    join dashboard d on c.subject_id = d.subject_id
    where d.viewposition in ('PA', 'AP')
) 
select  subject_id,
        anchor_time,
        dicom_id,
        path_image,
        round(time_diff::numeric, 2) AS time_diff_hours,
        target
from ranked_matches
where rank = 1  
  and time_diff <= 24.0
order by subject_id, anchor_time;
